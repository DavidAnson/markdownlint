  return input.replace(/[- ]([a-z])/g, "one"); // {MD039}
}

function MoreCodeButNotCode(input) {
  input = input.replace(/[- ]([a-z])/g, "two"); // {MD039}
  input = input.toLowerCase();
  input = input.replace(/[- ]([a-z])/g, "three"); // {MD039}
  return input;
}
